package ec.edu.uce;

public class Main {
    public static void main(String[] args) {
        Fecha fecha = new Fecha(29, 2, 2020); // Fecha válida (año bisiesto)

        if (Validaciones.validarFecha(fecha)) {
            System.out.println("La fecha " + fecha + " es válida.");
        } else {
            System.out.println("La fecha " + fecha + " no es válida.");
        }

        Fecha fechaInvalida = new Fecha(31, 4, 2021); // Fecha inválida (abril tiene 30 días)

        if (Validaciones.validarFecha(fechaInvalida)) {
            System.out.println("La fecha " + fechaInvalida + " es válida.");
        } else {
            System.out.println("La fecha " + fechaInvalida + " no es válida.");
        }
    }
}
